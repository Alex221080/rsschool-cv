from dataclasses import dataclass
from random import choice, shuffle


@dataclass()
class Card:
    rank: str
    suit: str


class Deck:
    suites = [b'\xE2\x99\xA0'.decode(),  # SPADE
              b'\xE2\x99\xA3'.decode(),  # CLUB
              b'\xE2\x99\xA6'.decode(),  # DIAMOND
              b'\xE2\x99\xA5'.decode()]  # HEART
    deck_types = {'Russian': 6,
                  'French': 2}

    def __init__(self, deck_type: str):
        try:
            self.deck_type = deck_type
            self.ranks = [str(i) for i in range(self.deck_types[deck_type], 11)] + ['J', 'Q', 'K', 'A']
            self.deck = [Card(rank, suit) for suit in self.suites
                         for rank in self.ranks]
        except KeyError:
            self.deck = []

    def __len__(self):
        return len(self.deck)

    def __getitem__(self, item):
        return self.deck[item]

    def __str__(self):
        if self.deck:
            return f' Deck created with {len(self.deck)} cards'
        else:
            return f'Deck must be only "Russian" or "French" not {self.deck_type}'


russian_deck = Deck('Russian')
print(russian_deck)
for card in russian_deck:
    print(card.rank + card.suit, end=' ')
print('\n')

french_deck = Deck('French')
print(french_deck)
for card in french_deck:
    print(card.rank + card.suit, end=' ')
print('\n')
